# công nghệ sử dụng
Lavarel Framework
PHP
Database: mysql

# Tạo một dự án
composer create-project --prefer-dist laravel/laravel laravel-project
# Sử dụng Breeze cho dự án
-Cài đặt Breeze thông qua Composer

- composer require laravel/breeze --dev
- php artisan breeze:install

- npm install
- npm run dev
